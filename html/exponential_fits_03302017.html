def exp_model(x, data):
    h = x[0]
    xx = np.zeros(len(data))
    rho = np.zeros(len(data))
    for i in range(len(data)):
        xx[i] = data[i][0]
        rho[i] = data[i][1]
    
    rho_model = rho.max() - h*xx
    
    chisq = np.sum( (rho-rho_model)**2 )

    return chisq
